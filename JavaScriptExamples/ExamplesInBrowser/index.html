Uses the closure pattern to create a timer that can be bound to a DOM element. 

function makeClosureForTimer(element, interval) {

  var counter = 0; // Inner variable to this function.  Not visible outside.

  setInterval(timeIt, interval);

  // This function will be called on a timer.  It uses the counter to update the DOM periodically.
  function timeIt() {
    element.textContent = `${interval}ms per tick: ${counter}`;
    counter++;
  }
}
